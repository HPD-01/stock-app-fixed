import { serve } from "https://deno.land/std@0.224.0/http/server.ts";
import { createClient } from "https://esm.sh/@supabase/supabase-js@2";

serve(async (req) => {
  const SUPABASE_URL = Deno.env.get("SUPABASE_URL")!;
  const SUPABASE_SERVICE_ROLE_KEY = Deno.env.get("SUPABASE_SERVICE_ROLE_KEY")!;
  const MAILER_WEBHOOK_URL = Deno.env.get("MAILER_WEBHOOK_URL")!;
  const ALERT_TO_EMAIL = Deno.env.get("ALERT_TO_EMAIL")!;

  const payload = await req.json();

  if (payload.type !== "INSERT") {
    return new Response("ignored");
  }

  const itemId = payload.record?.item_id;
  const qty = payload.record?.quantity ?? payload.record?.qty;

  if (!itemId) {
    return new Response("no item");
  }

  const supabase = createClient(
    SUPABASE_URL,
    SUPABASE_SERVICE_ROLE_KEY
  );

  const { data: item } = await supabase
    .from("items")
    .select("sku, name, reorder_threshold")
    .eq("id", itemId)
    .single();

  if (!item) {
    return new Response("item not found", { status: 404 });
  }

  await fetch(MAILER_WEBHOOK_URL, {
    method: "POST",
    headers: { "content-type": "application/json" },
    body: JSON.stringify({
      to: ALERT_TO_EMAIL,
      sku: item.sku,
      name: item.name,
      qty,
      threshold: item.reorder_threshold,
    }),
  });

  return new Response("ok");
});
